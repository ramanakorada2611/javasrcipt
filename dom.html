9.Write code to implement timer clock using JS -- display HH:MM:SS -- the time should keep updating every second

Ans:
    function clock(){
       let time=new Date()
       let hrs=time.getHours()
       let mins=time.getMinutes()
       let sec=time.getSeconds()
       let tag=document.getElementById("clock")
       period="AM"
       if(hrs>=12){
        period="PM"
       }
       if(hrs>12){
          hrs=hrs%12
       }
       if(hrs<10){
          hrs="0"+hrs
       }
       if(mins < 10){
          mins="0"+mins
       }
       if(sec < 10){
          sec="0"+sec
       }
       tag.innerText=+hrs+ " : " +mins+ " : " +sec+" : " +period;
    }
    setInterval(clock, 1000);
